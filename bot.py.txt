import requests
import time
from telegram.ext import Updater, CommandHandler

API_KEY = "ISI_TOKEN_BOT_TELEGRAM_KAMU"
TEMPMAIL_DOMAIN = "https://api.mail.tm"

def create_email():
    r = requests.post(f"{TEMPMAIL_DOMAIN}/accounts", json={
        "address": "",
        "password": "password"
    })
    return r.json()

def get_messages(token):
    headers = {"Authorization": f"Bearer {token}"}
    r = requests.get(f"{TEMPMAIL_DOMAIN}/messages", headers=headers)
    return r.json()

def start(update, context):
    update.message.reply_text("Bot aktif! Ketik /mail untuk buat email baru.")

def mail(update, context):
    acct = create_email()

    if "address" not in acct:
        update.message.reply_text("Gagal membuat email.")
        return

    email = acct["address"]
    token = acct["token"]
    update.message.reply_text(f"Email baru:\n\nðŸ“§ {email}\n\nTunggu OTP masuk otomatis...")

    # cek inbox tiap 5 detik
    while True:
        msgs = get_messages(token)
        if msgs["hydra:member"]:
            msg = msgs["hydra:member"][0]
            text = msg["text"]
            update.message.reply_text(f"ðŸ“© Pesan diterima:\n\n{text}")
            break
        time.sleep(5)

def main():
    updater = Updater(API_KEY, use_context=True)
    dp = updater.dispatcher

    dp.add_handler(CommandHandler("start", start))
    dp.add_handler(CommandHandler("mail", mail))

    updater.start_polling()
    updater.idle()

if __name__ == "__main__":
    main()
